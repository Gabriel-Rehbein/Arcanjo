document.addEventListener("DOMContentLoaded", async () => {
    console.log("🚀 Chatbot Inteligente Pronto!");

    const inputText = document.getElementById("inputText");
    const sendButton = document.getElementById("sendButton");
    const chatBox = document.getElementById("chat-box");
    const voiceToggle = document.getElementById("voiceToggle");

    if (!inputText || !sendButton || !chatBox || !voiceToggle) {
        console.error("❌ Erro: Elementos do chat não encontrados.");
        return;
    }

    let chatHistory = [];
    let knowledgeDB = {};
    let AI_API_KEY = "SUA_CHAVE_OPENAI";
    let personality = "inteligente e amigável";

    await loadKnowledge();

    sendButton.addEventListener("click", () => {
        console.log("Botão de enviar clicado");
        processMessage();
    });

    inputText.addEventListener("keypress", (event) => {
        console.log(`Tecla pressionada: ${event.key}`);
        if (event.key === "Enter") {
            event.preventDefault();
            processMessage();
        }
    });

    async function processMessage() {
        const message = inputText.value.trim();
        if (!message) return;

        appendMessage("user", message);
        inputText.value = "";

        appendLoadingMessage();

        try {
            let response = await generateResponse(message);
            updateLastBotMessage(response);
            if (voiceToggle.checked) speak(response);
            chatHistory.push({ user: message, bot: response });
            storeKnowledge(message, response);
        } catch (error) {
            console.error("❌ Erro ao gerar resposta:", error);
            updateLastBotMessage("⚠️ Ocorreu um erro ao processar a resposta.");
        }
    }

    function appendMessage(sender, text) {
        const messageElement = document.createElement("div");
        messageElement.classList.add("message", sender === "user" ? "user-message" : "bot-message");
        messageElement.textContent = text;
        chatBox.appendChild(messageElement);
        scrollToBottom();
    }

    function appendLoadingMessage() {
        const loadingElement = document.createElement("div");
        loadingElement.classList.add("message", "bot-message", "loading-message");
        loadingElement.textContent = "⏳ Processando...";
        chatBox.appendChild(loadingElement);
        scrollToBottom();
    }

    function updateLastBotMessage(text) {
        const botMessages = chatBox.getElementsByClassName("bot-message");
        if (botMessages.length > 0) {
            botMessages[botMessages.length - 1].textContent = text;
        }
    }

    function speak(text) {
        const synth = window.speechSynthesis;
        synth.cancel();
        const utterance = new SpeechSynthesisUtterance(text);
        utterance.lang = "pt-BR";
        utterance.rate = 1.1;
        utterance.pitch = 1.2;
        synth.speak(utterance);
    }

    async function generateResponse(question) {
        question = sanitizeText(question);
        if (knowledgeDB[question]) return knowledgeDB[question];

        let bestMatch = findBestMatch(question);
        if (bestMatch.similarity > 0.7) return bestMatch.response;

        return await searchAI(question);
    }

    async function searchAI(query) {
        try {
            const response = await fetch("https://api.openai.com/v1/chat/completions", {
                method: "POST",
                headers: {
                    "Authorization": `Bearer ${AI_API_KEY}`,
                    "Content-Type": "application/json"
                },
                body: JSON.stringify({
                    model: "gpt-4-turbo",
                    messages: [
                        { role: "system", content: `Você é um assistente com personalidade ${personality}.` },
                        { role: "user", content: query }
                    ],
                    max_tokens: 200,
                    temperature: 0.7
                })
            });

            const data = await response.json();
            if (!response.ok) {
                throw new Error(data.error?.message || "Erro desconhecido na API da OpenAI");
            }
            return data.choices?.[0]?.message?.content.trim() || "🤖 Não encontrei uma resposta exata, mas posso aprender se você me ensinar.";
        } catch (error) {
            console.error("❌ Erro na busca AI:", error);
            return "⚠️ Erro ao buscar resposta na IA.";
        }
    }

    async function loadKnowledge() {
        knowledgeDB = await getAllFromIndexedDB("chatbotKnowledge") || {};
    }

    async function getAllFromIndexedDB(storeName) {
        return new Promise((resolve, reject) => {
            let request = indexedDB.open("ChatbotDB", 1);
            request.onsuccess = (event) => {
                let db = event.target.result;
                let transaction = db.transaction(storeName, "readonly");
                let store = transaction.objectStore(storeName);
                let data = {};
                let cursorRequest = store.openCursor();
                cursorRequest.onsuccess = (event) => {
                    let cursor = event.target.result;
                    if (cursor) {
                        data[cursor.key] = cursor.value;
                        cursor.continue();
                    } else {
                        resolve(data);
                    }
                };
                cursorRequest.onerror = () => reject(cursorRequest.error);
            };
            request.onerror = () => reject(request.error);
        });
    }

    function sanitizeText(text) {
        return text.toLowerCase().replace(/[^\w\s]/gi, "").trim();
    }

    function scrollToBottom() {
        setTimeout(() => {
            chatBox.scrollTop = chatBox.scrollHeight;
        }, 100);
    }
});
